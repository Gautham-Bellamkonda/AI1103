\section{Problem}
Let $X_1, X_2, \ldots , X_n$ be a random sample of size $n \; ( \geq 2 ) $ from a distribution having the probability density function
\begin{align}
f(x;\theta) = 
\begin{cases}
\dfrac{1}{\theta} \exp ( -\frac{x}{\theta}) & x > 0, \\
0, & \text{otherwise,}
\end{cases}
\end{align}
where $\theta \in (0, \infty)$. Let $X_{(1)} = $ min $ \{ X_1, X_2, \ldots , X_n \} $ and $T = \sum_{i=1}^n X_i $. Then $E(X_{(1)} | T)$ equals $\ldots \ldots \ldots$
\begin{itemize}
\item $\dfrac{T}{n^2}$ \\
\item $\dfrac{T}{n}$ \\
\item $\dfrac{(n+1)T}{2n}$ \\
\item $\dfrac{(n+1)^2 T}{4n^2}$
\end{itemize}
\section{Solution}
For $n = 2$, 
\begin{align}
E(X_{(1)} | T) &= \int_{ - \infty }^{ \infty } x f_{ X_{(1)} }( x | T ) dx \\
&= \int_{ - \infty }^{ \infty } x \frac{f_{X_1 , X_2}(x, T-x) + f_{X_1 , X_2}(T-x, x)}{2 f_{X_1 + X_2}(T)} dx\\
&= \int_{0}^{T} x  \frac{\exp(-\frac{x}{\theta})\exp(-\frac{T-x}{\theta})}{\theta^2 ( \frac{1}{\theta^2} T \exp(-\frac{T}{\theta})) }  dx \\
&= \int_{0}^{T} x  \frac{1}{T}  dx \\
&= \frac{1}{T} \int_{0}^{T} x dx \\
&= \frac{T}{2}
\end{align}
The given key is $\dfrac{T}{n^2} = \dfrac{T}{4}$
